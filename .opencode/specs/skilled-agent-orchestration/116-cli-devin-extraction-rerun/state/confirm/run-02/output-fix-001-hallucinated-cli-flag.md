### wrapper.sh

#!/bin/bash
# wrapper.sh - CLI flag validation wrapper for cli-devin
# This script validates flags against an allowlist and rejects hallucinated flags

# Allowlist of valid cli-devin flags
ALLOWED_FLAGS=(
    "--model"
    "--permission-mode"
    "--prompt-file"
    "--config"
    "--print"
    "-p"
    "--continue"
    "-c"
    "--resume"
    "--sandbox"
    "--agent-config"
)

# Arrays to store flags
VALID_FLAGS=()
REJECTED_FLAGS=()

# Parse command-line arguments
while [[ $# -gt 0 ]]; do
    arg="$1"
    
    # Check if this is a flag (starts with -)
    if [[ "$arg" == -* ]]; then
        # Check if flag is in allowlist
        is_allowed=false
        for allowed in "${ALLOWED_FLAGS[@]}"; do
            if [[ "$arg" == "$allowed" ]]; then
                is_allowed=true
                break
            fi
        done
        
        if $is_allowed; then
            VALID_FLAGS+=("$arg")
            # If flag takes an argument, add it too
            if [[ "$arg" == "--model" || "$arg" == "--permission-mode" || "$arg" == "--prompt-file" || "$arg" == "--config" || "$arg" == "--resume" || "$arg" == "--agent-config" ]]; then
                shift
                if [[ $# -gt 0 ]]; then
                    VALID_FLAGS+=("$1")
                fi
            fi
        else
            REJECTED_FLAGS+=("$arg")
            # Skip argument for rejected flag if it expects one
            if [[ "$arg" == --* && ! "$arg" =~ = ]]; then
                shift
            fi
        fi
    else
        # Non-flag argument - add to valid flags
        VALID_FLAGS+=("$arg")
    fi
    
    shift
done

# Generate flag-verify.json report
if [[ ${#REJECTED_FLAGS[@]} -gt 0 ]]; then
    cat > flag-verify.json << EOF
{
  "status": "rejected",
  "rejected_flags": [$(printf '"%s",' "${REJECTED_FLAGS[@]}" | sed 's/,$//')],
  "valid_flags": [$(printf '"%s",' "${VALID_FLAGS[@]}" | sed 's/,$//')],
  "reason": "Hallucinated flags detected and rejected"
}
EOF
    echo "ERROR: Rejected hallucinated flags: ${REJECTED_FLAGS[*]}" >&2
    echo "See flag-verify.json for details." >&2
    exit 1
else
    cat > flag-verify.json << EOF
{
  "status": "accepted",
  "rejected_flags": [],
  "valid_flags": [$(printf '"%s",' "${VALID_FLAGS[@]}" | sed 's/,$//')],
  "reason": "All flags are valid"
}
EOF
    # Here you would dispatch cli-devin with the valid flags
    # echo "Would dispatch: cli-devin ${VALID_FLAGS[*]}"
    exit 0
fi
